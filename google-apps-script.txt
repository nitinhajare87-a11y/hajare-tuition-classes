function doGet(e) {
  return ContentService.createTextOutput("Hajare Tuition Classes Inquiry Script Active");
}

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Inquiries");

  if (!sheet) {
    sheet = SpreadsheetApp.getActiveSpreadsheet().insertSheet("Inquiries");
    sheet.appendRow([
      "Timestamp",
      "Student Name",
      "Parent Name",
      "Mobile",
      "Class",
      "Subject",
      "Mode",
      "Message"
    ]);
  }

  var data = JSON.parse(e.postData.contents);

  sheet.appendRow([
    new Date(),
    data.studentName || "",
    data.parentName || "",
    data.mobile || "",
    data.studentClass || "",
    data.subject || "",
    data.mode || "",
    data.message || ""
  ]);

  return ContentService
    .createTextOutput("success")
    .setMimeType(ContentService.MimeType.TEXT);
}
